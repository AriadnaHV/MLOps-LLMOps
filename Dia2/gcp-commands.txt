
docker build -t gcr.io/"RESOURCE_NAME"/"IMAGE_NAME" .


NOTE: En caso de que no tengamos ninguna carpeta para las imagen primero tenemos que crearla.

gcloud builds submit --tag "REGION"-docker.pkg.dev/"RESOURCE_NAME"/"ARTIFACT_FOLDER_NAME"/"IMAGE_NAME":latest .

##Si al ejecutar el gcloud builds nos da un error por permisos tenemos que ejecutar el siguiente comando: 
gcloud auth configure-docker "REGION"-docker.pkg.dev

gcloud run deploy "IMAGE_NAME" \
  --image "REGION"-docker.pkg.dev/"RESOURCE_NAME"/:ARTIFACT_FOLDER_NAME"/"IMAGE_NAME":latest \
  --platform managed \
  --region "REGION" \
  --allow-unauthenticated
